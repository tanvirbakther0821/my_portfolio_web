3. Data Processing Module

data_processing.py
import csv

def username():
    return 'your_username'

def data_wrangling(filter_class=None):
    """
    Process CSV data and return formatted results
    
    Args:
        filter_class (str): Optional filter for data category
    
    Returns:
        tuple: (header, table_data, dropdown_options)
    """
    with open('data/dataset.csv', 'r', encoding='utf-8') as f:
        reader = csv.reader(f)
        table = []
        all_classes = set()
        
        # Read header row
        header = next(reader)
        
        # Read and process data rows
        for row in reader:
            # Convert numeric columns
            processed_row = [
                row[0],                    # species (string)
                row[1],                    # class (string) 
                int(row[2]) if row[2] else 0  # count (integer)
            ]
            table.append(processed_row)
            all_classes.add(row[1])  # Collect unique classes
        
        # Create sorted dropdown options
        dropdown_options = sorted(list(all_classes))
        
        # Apply filtering if specified
        if filter_class:
            table = [row for row in table if row[1] == filter_class]
        
        # Sort by count (descending) and limit to top 10
        table.sort(key=lambda x: x[2], reverse=True)
        table = table[:10]
    
    return header, table, dropdown_options

from flask import render_template, request

@app.route('/')

    """Main page route"""
    # Get filter parameter from URL

@app.route('/api/data')
def api_data():
    """API endpoint for JSON data"""

        'data': table,
        'filters': dropdown_options,
        'count': len(table)
    }

RegEx Fundamentals

Basic Syntax Elements

Pattern Description Example Matches
. Any single character a.c abc, axc, a1c
* Zero or more of preceding ab*c ac, abc, abbc
+ One or more of preceding ab+c abc, abbc (not ac)
? Zero or one of preceding ab?c ac, abc
^ Start of string ^Hello Hello world
$ End of string world$ Hello world
\b Word boundary \bcat\b cat (not category)

Character Classes

Class Description Equivalent
[a-z] Lowercase letters a, b, c, ..., z
[A-Z] Uppercase letters A, B, C, ..., Z
[0-9] Digits 0, 1, 2, ..., 9
\d Digit [0-9]
\w Word character [a-zA-Z0-9_]
\s Whitespace Space, tab, newline

Advanced RegEx Techniques

Lookahead and Lookbehind

Advanced Patterns
# Positive lookahead: match word followed by "Airport"
pattern = r'\b\w+(?=\s+Airport)'
text = "Beijing Capital Airport and Shanghai Pudong Airport"
matches = re.findall(pattern, text)
print(matches)  # ['Capital', 'Pudong']

# Positive lookbehind: match word preceded by "Airport"
pattern = r'(?<=Airport\s)\w+'
text = "Airport Code: LAX, Airport Name: Los Angeles"
matches = re.findall(pattern, text)
print(matches)  # ['Code:', 'Name:']

Named Groups

Named Capture Groups
# Extract structured data with named groups
pattern = r'(?P<city>[A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)\s+(?P<type>(?:International\s+)?Airport)'

text = "Flights to Beijing Capital International Airport"
match = re.search(pattern, text)
if match:
    city = match.group('city')      # "Beijing Capital"
    airport_type = match.group('type')  # "International Airport"
    print(f"City: {city}, Type: {airport_type}")

import re

def clean_airport_data(text):
    """Comprehensive airport name extraction and cleaning"""
    if not text or pd.isna(text):
        return ""
    
    # Remove common prefixes
    text = re.sub(r'^(seized\s+at|at|from|to|in)\s+', '', text, flags=re.IGNORECASE)
    
    # Extract airport name patterns
    patterns = [
        r'\b([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)\s+(International\s+Airport)\b',
        r'\b([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)\s+(Airport)\b'
    ]
    
    for pattern in patterns:
        match = re.search(pattern, text)
        if match:
            city_part = match.group(1)
            airport_part = match.group(2)
            
            # Skip common non-city words
            if city_part not in ['The', 'At', 'From', 'To', 'In', 'Seized']:
                return f"{city_part} {airport_part}"
    
    return ""

# Usage example
texts = [
    "77kg of sandalwood at the Chhatrapati Shivaji Maharaj International Airport",
    "Ivory seized at Beijing Capital Airport",
    "Live animals found at Chennai International Airport"
]

for text in texts:
    airport = clean_airport_data(text)
    print(f"'{text}' ‚Üí '{airport}'")

Data Validation Patterns

Data Quality Checks
def validate_dataset(df):
    """Comprehensive data validation"""
    validation_results = {}
    
    # Check for required columns
    required_cols = ['ID', 'Country', 'Date', 'Subject']
    missing_cols = [col for col in required_cols if col not in df.columns]
    validation_results['missing_columns'] = missing_cols
    
    # Check for empty values
    empty_counts = df.isnull().sum()
    validation_results['empty_values'] = empty_counts[empty_counts > 0].to_dict()
    
    # Check for duplicates
    duplicate_count = df.duplicated().sum()
    validation_results['duplicates'] = duplicate_count
    
    # Check data types
    validation_results['data_types'] = df.dtypes.to_dict()
    
    # Check value ranges
    numeric_cols = df.select_dtypes(include=[np.number]).columns
    for col in numeric_cols:
        validation_results[f'{col}_range'] = {
            'min': df[col].min(),
            'max': df[col].max(),
            'negative_count': (df[col] < 0).sum()
        }
    
    return validation_results

Debug RegEx Pattern
import re

def debug_regex(pattern, text):
    """Debug regex pattern with detailed output"""
    print(f"Pattern: {pattern}")
    print(f"Text: {text}")
    
    match = re.search(pattern, text)
    if match:
        print(f"Full match: '{match.group()}'")
        for i, group in enumerate(match.groups(), 1):
            print(f"Group {i}: '{group}'")
    else:
        print("No match found")
    
    # Test with findall for multiple matches
    all_matches = re.findall(pattern, text)
    print(f"All matches: {all_matches}")

# Usage
debug_regex(r'([A-Z][a-z]+)\s+(Airport)', 
           "Beijing Capital Airport and Shanghai Pudong Airport")
